-- CREACION TABLA COCHES
CREATE TABLE COCHES(
    ID INTEGER AUTO_INCREMENT,
    MATRICULA VARCHAR(50) NOT NULL,
    MARCA VARCHAR(50) NOT NULL,
    MODELO VARCHAR(50) NOT NULL,
    COLOR VARCHAR(50) NOT NULL,
    PRIMARY KEY (ID)
);

-- CREACION TABLA PASAJEROS
CREATE TABLE PASAJEROS(
    ID INTEGER AUTO_INCREMENT,
    NOMBRE VARCHAR(50) NOT NULL,
    EDAD INTEGER NOT NULL,
    PESO FLOAT NOT NULL,
    PRIMARY KEY (ID)
);

-- CREACION TABLA COCHES_PASAJEROS
CREATE TABLE COCHES_PASAJEROS(
    IDCOCHE INTEGER,
    IDPASAJERO INTEGER,
    FOREIGN KEY(IDCOCHE) REFERENCES COCHES(ID),
    FOREIGN KEY(IDPASAJERO) REFERENCES PASAJEROS(ID),
    PRIMARY KEY(IDCOCHE, IDPASAJERO)
);
